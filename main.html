def neural_network(input, weights):
    out = 0
    for i in range(len(input)):
        out += input[i] * weights[i]
    return out

def ele_mul(scaler, vector):
    out = [0, 0, 0]
    for i in range(len(vector)):
        out[i] = scaler * vector[i]
    return out

    output = [0, 0, 0]

        for j in range(len(matrix[0])):
